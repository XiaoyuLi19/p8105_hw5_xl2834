``` r
sim_mean_sd = function(n = 30, mu = 0, sigma = 5) {
  
  sim_data = tibble(
    x = rnorm(n, mean = mu, sd = sigma),
  )
  
  sim_data %>% 
    summarize(
      mu_hat = mean(x),
      p_value = broom::tidy(t.test(x, mu = 0))$p.value
    )
}


sim_results = 
  rerun(50, sim_mean_sd(30, 0, 5)) %>% 
  bind_rows()
```
